// ============================================================================
// 粒子生命周期状态（互斥，粒子同一时刻只能处于一种状态）
// ============================================================================
enum class ParticleStatus : std::uint8_t
{
    // --- 初始态 ---
    UNINITIALIZED   = 0,   // 未初始化
    CREATED         = 1,   // 已创建，等待注入模拟
    
    // --- 活跃态 ---
    ACTIVE          = 10,  // 正在被追踪
    SUSPENDED       = 11,  // 暂停追踪（等待后续处理）
    WAITING         = 12,  // 等待队列中（多线程调度）
    
    // --- 终止态（粒子生命结束）---
    ABSORBED        = 20,  // 被材料完全吸收
    ESCAPED         = 21,  // 逃逸出模拟边界
    DEAD            = 22,  // 能量低于截断阈值
    LOST            = 23,  // 权重过低被俄罗斯轮盘赌淘汰
    DECAYED         = 24,  // 粒子衰变
    ANNIHILATED     = 25,  // 湮灭（如正负电子对湮灭）
    CAPTURED        = 26,  // 被俘获（如中子俘获）
    FUSED           = 27,  // 聚变
    FISSIONED       = 28,  // 裂变
    THERMALIZED     = 29,  // 热化（能量降至热平衡）
    STUCK           = 30,  // 卡住（数值问题，无法继续）
    TIME_EXCEEDED   = 31,  // 超过最大模拟时间
    STEP_EXCEEDED   = 32,  // 超过最大步数限制
    
    // --- 错误态 ---
    ERROR           = 250, // 发生错误
    INVALID         = 255, // 无效状态
};

// ============================================================================
// 模拟标志（位掩码，可组合）
// ============================================================================
enum class SimulationFlags : std::uint32_t
{
    NONE                = 0,
    
    // --- 物理过程开关 ---
    ENABLE_IONIZATION       = 1 << 0,
    ENABLE_EXCITATION       = 1 << 1,
    ENABLE_BREMSSTRAHLUNG   = 1 << 2,
    ENABLE_PAIR_PRODUCTION  = 1 << 3,
    ENABLE_PHOTOELECTRIC    = 1 << 4,
    ENABLE_COMPTON          = 1 << 5,
    ENABLE_RAYLEIGH         = 1 << 6,
    ENABLE_NUCLEAR          = 1 << 7,
    ENABLE_ELASTIC          = 1 << 8,
    ENABLE_FLUORESCENCE     = 1 << 9,
    ENABLE_AUGER            = 1 << 10,
    
    // --- 追踪选项 ---
    TRACK_SECONDARIES       = 1 << 16,
    TRACK_OPTICAL_PHOTONS   = 1 << 17,
    TRACK_LOW_ENERGY        = 1 << 18,
    
    // --- 方差缩减 ---
    USE_SPLITTING           = 1 << 20,
    USE_ROULETTE            = 1 << 21,
    USE_WEIGHT_WINDOW       = 1 << 22,
    USE_IMPORTANCE_SAMPLING = 1 << 23,
    
    // --- 记录选项 ---
    RECORD_TRAJECTORY       = 1 << 24,
    RECORD_ENERGY_DEPOSIT   = 1 << 25,
    RECORD_INTERACTIONS     = 1 << 26,
    RECORD_BOUNDARY_CROSS   = 1 << 27,
    
    // --- 常用组合 ---
    ALL_EM_PHYSICS          = ENABLE_IONIZATION | ENABLE_EXCITATION | 
                              ENABLE_BREMSSTRAHLUNG | ENABLE_PHOTOELECTRIC |
                              ENABLE_COMPTON | ENABLE_RAYLEIGH,
    
    DEFAULT                 = ENABLE_IONIZATION | ENABLE_EXCITATION |
                              ENABLE_ELASTIC | TRACK_SECONDARIES,
};

// 位运算支持
inline SimulationFlags operator|(SimulationFlags a, SimulationFlags b) {
    return static_cast<SimulationFlags>(
        static_cast<std::uint32_t>(a) | static_cast<std::uint32_t>(b));
}

inline SimulationFlags operator&(SimulationFlags a, SimulationFlags b) {
    return static_cast<SimulationFlags>(
        static_cast<std::uint32_t>(a) & static_cast<std::uint32_t>(b));
}

inline bool hasFlag(SimulationFlags flags, SimulationFlags flag) {
    return (flags & flag) == flag;
}

// ============================================================================
// 辅助函数：状态判断
// ============================================================================
inline bool isAlive(ParticleStatus status) {
    return status == ParticleStatus::ACTIVE || 
           status == ParticleStatus::SUSPENDED ||
           status == ParticleStatus::WAITING;
}

inline bool isTerminated(ParticleStatus status) {
    auto s = static_cast<std::uint8_t>(status);
    return s >= 20 && s < 200;
}

inline bool isError(ParticleStatus status) {
    return status == ParticleStatus::ERROR || 
           status == ParticleStatus::INVALID ||
           status == ParticleStatus::STUCK;
}

// ============================================================================
// 辅助函数：转为字符串（用于调试/日志）
// ============================================================================
inline const char* toString(ParticleStatus status) {
    switch (status) {
        case ParticleStatus::UNINITIALIZED: return "UNINITIALIZED";
        case ParticleStatus::CREATED:       return "CREATED";
        case ParticleStatus::ACTIVE:        return "ACTIVE";
        case ParticleStatus::SUSPENDED:     return "SUSPENDED";
        case ParticleStatus::WAITING:       return "WAITING";
        case ParticleStatus::ABSORBED:      return "ABSORBED";
        case ParticleStatus::ESCAPED:       return "ESCAPED";
        case ParticleStatus::DEAD:          return "DEAD";
        case ParticleStatus::LOST:          return "LOST";
        case ParticleStatus::DECAYED:       return "DECAYED";
        case ParticleStatus::ANNIHILATED:   return "ANNIHILATED";
        case ParticleStatus::CAPTURED:      return "CAPTURED";
        case ParticleStatus::FUSED:         return "FUSED";
        case ParticleStatus::FISSIONED:     return "FISSIONED";
        case ParticleStatus::THERMALIZED:   return "THERMALIZED";
        case ParticleStatus::STUCK:         return "STUCK";
        case ParticleStatus::TIME_EXCEEDED: return "TIME_EXCEEDED";
        case ParticleStatus::STEP_EXCEEDED: return "STEP_EXCEEDED";
        case ParticleStatus::ERROR:         return "ERROR";
        case ParticleStatus::INVALID:       return "INVALID";
        default:                            return "UNKNOWN";
    }
}

inline const char* toString(ParticleType type) {
    switch (type) {
        case ParticleType::ELECTRON:    return "electron";
        case ParticleType::POSITRON:    return "positron";
        case ParticleType::PHOTON:      return "photon";
        case ParticleType::GAMMA:       return "gamma";
        case ParticleType::PROTON:      return "proton";
        case ParticleType::NEUTRON:     return "neutron";
        case ParticleType::ALPHA:       return "alpha";
        case ParticleType::HEAVY_ION:   return "heavy_ion";
        case ParticleType::PHONON:      return "phonon";
        case ParticleType::HOLE:        return "hole";
        default:                        return "unknown";
    }
}
